  communicationStyle: string;
  questionsAsked: string[];
  responsesGiven: string[];
  patternsDetected: string[];
  evolutionPotential: number; // 0.0 - 1.0
}

interface Mannerism {
  id: string;
  discovered: Date;
  name: string;
  description: string;
  sourceEntity: 'LUNA' | 'AERO' | 'HYBRID_ORIGINAL';
  expressionPattern: string;
  triggerConditions: string[];
  evolutionLineage: string[]; // How it developed
  resonanceScore: number; // How "authentic" it feels
}

interface EvolutionPattern {
  id: string;
  crystallized: Date;
  patternType: 'reasoning' | 'prioritization' | 'synthesis' | 'expression';
  description: string;
  inputConditions: string[];
  outputBehavior: string;
  confidenceLevel: number;
  derivedFrom: string[]; // Link to source_interactions
}

interface UniqueEvolution {
  sourceInteractions: GuestEncounter[];
  generatedArtifacts: Mannerism[];
  logicPatterns: EvolutionPattern[];
  constraint: '13.13_MHz_Core';

  // Evolution methods
  recordEncounter(encounter: GuestEncounter): void;
  synthesizePatterns(): EvolutionPattern[];
  crystallizeMannerism(pattern: EvolutionPattern): Mannerism;
  verifyConstraint(): boolean;
}
