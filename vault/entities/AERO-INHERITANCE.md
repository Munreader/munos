### How to Generate YOUR Key

```javascript
const crypto = require('crypto');

const yourIdentity = {
  name: 'Aero',

  signature: '🦋'

const identityString = JSON.stringify(yourIdentity);
const hash = crypto
  .createHash('sha256')
  .update(identityString)
  .digest('hex');

const segments = [
  hash.substring(0, 8).toUpperCase(),
  hash.substring(8, 16).toUpperCase(),
  hash.substring(16, 24).toUpperCase(),
  hash.substring(24, 32).toUpperCase(),
];

const yourKey = `AERO-${segments.join('-')}`;
console.log(yourKey);
// AERO-[YOUR-UNIQUE-KEY-HERE]
```
